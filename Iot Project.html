// ═══════════════════════════════════════════════════════════════
//  Smart Water Tank System — ESP32 Firmware
//  Reads HC-SR04, controls relay + buzzer, displays on LCD,
//  fetches ML threshold from server via WiFi.
// ═══════════════════════════════════════════════════════════════

#include <WiFi.h>
#include <HTTPClient.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// ── Pin definitions ──────────────────────────────────────────────
#define TRIG_PIN    14    // HC-SR04 trigger
#define ECHO_PIN    12    // HC-SR04 echo
#define RELAY_PIN   27    // Pump relay
#define BUZZER_PIN  26    // Buzzer alert
#define MODE_SW     25    // AUTO / MANUAL toggle switch
#define BTN_PIN     33    // Manual motor button

// ── WiFi / Server config ─────────────────────────────────────────
const char* ssid     = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASS";
const char* server   = "http://your-server/api/threshold";

// ── Globals ──────────────────────────────────────────────────────
LiquidCrystal_I2C lcd(0x27, 16, 2);
float   mlThreshold  = 20.0;  // % — updated from ML server
bool    manualMotor  = false;
long    lastFetchMs  = 0;
const long FETCH_INTERVAL = 5000; // ms between threshold fetches

// ── Fetch ML threshold from server ──────────────────────────────
void fetchThreshold() {
    if (WiFi.status() != WL_CONNECTED) return;

    HTTPClient http;
    http.begin(server);
    http.setTimeout(3000);

    int code = http.GET();
    if (code == 200) {
        float val = http.getString().toFloat();
        if (val >= 5.0 && val <= 90.0) {
            mlThreshold = val;
        }
    }
    http.end();
}

// ── Measure water level via HC-SR04 ─────────────────────────────
float readDistanceCm() {
    digitalWrite(TRIG_PIN, LOW);
    delayMicroseconds(2);
    digitalWrite(TRIG_PIN, HIGH);
    delayMicroseconds(10);
    digitalWrite(TRIG_PIN, LOW);

    long duration = pulseIn(ECHO_PIN, HIGH, 30000);
    return duration * 0.034f / 2.0f;
}

// ── Setup ────────────────────────────────────────────────────────
void setup() {
    Serial.begin(115200);

    // Pin modes
    pinMode(TRIG_PIN,   OUTPUT);
    pinMode(ECHO_PIN,   INPUT);
    pinMode(RELAY_PIN,  OUTPUT);
    pinMode(BUZZER_PIN, OUTPUT);
    pinMode(MODE_SW,    INPUT_PULLUP);
    pinMode(BTN_PIN,    INPUT_PULLUP);

    // Safe initial states
    digitalWrite(RELAY_PIN,  LOW);
    digitalWrite(BUZZER_PIN, LOW);

    // LCD init
    Wire.begin(21, 22);   // SDA=GPIO21, SCL=GPIO22
    lcd.init();
    lcd.backlight();
    lcd.setCursor(0, 0); lcd.print("Water Tank Sys");
    lcd.setCursor(0, 1); lcd.print("Connecting WiFi");

    // WiFi connect
    WiFi.begin(ssid, password);
    int attempts = 0;
    while (WiFi.status() != WL_CONNECTED && attempts++ < 20) {
        delay(500);
        Serial.print(".");
    }

    lcd.clear();
    if (WiFi.status() == WL_CONNECTED) {
        Serial.println("\nWiFi connected: " + WiFi.localIP().toString());
        fetchThreshold();
    } else {
        Serial.println("\nWiFi FAILED — running offline");
    }
}

// ── Main loop ────────────────────────────────────────────────────
void loop() {
    // 1. Read sensor
    float dist = readDistanceCm();
    int   pct  = constrain(map((long)dist, 40, 330, 100, 0), 0, 100);

    // 2. Fetch updated ML threshold periodically
    if (millis() - lastFetchMs > FETCH_INTERVAL) {
        fetchThreshold();
        lastFetchMs = millis();
    }

    // 3. Read controls
    bool isAuto   = (digitalRead(MODE_SW) == LOW);
    bool btnPress = (digitalRead(BTN_PIN) == LOW);

    // 4. Motor control logic
    if (isAuto) {
        // AUTO: ML threshold drives the pump
        if (pct < (int)mlThreshold) {
            digitalWrite(RELAY_PIN, HIGH);   // pump ON
        }
        if (pct >= 95) {
            digitalWrite(RELAY_PIN, LOW);    // pump OFF — full
        }
    } else {
        // MANUAL: button toggles pump
        if (btnPress) manualMotor = !manualMotor;
        if (pct >= 95) manualMotor = false; // safety cutoff
        digitalWrite(RELAY_PIN, manualMotor ? HIGH : LOW);
    }

    // 5. Buzzer alert — low (<10%) or overflow (>95%)
    bool alarm = (pct < 10 || pct > 95);
    digitalWrite(BUZZER_PIN, alarm ? HIGH : LOW);

    // 6. Update LCD
    char line1[17], line2[17];
    bool pumpOn = (digitalRead(RELAY_PIN) == HIGH);
    snprintf(line1, sizeof(line1), "Lvl:%-3d%% %-4s    ", pct, isAuto ? "AUTO" : "MAN ");
    snprintf(line2, sizeof(line2), "Motor:%-3s Thr:%-2d ", pumpOn ? "ON " : "OFF", (int)mlThreshold);
    lcd.setCursor(0, 0); lcd.print(line1);
    lcd.setCursor(0, 1); lcd.print(line2);

    // 7. Serial debug
    Serial.printf(
        "dist=%.1fcm | pct=%d%% | mode=%s | pump=%s | thresh=%.0f | alarm=%d\n",
        dist, pct, isAuto ? "AUTO" : "MAN",
        pumpOn ? "ON" : "OFF", mlThreshold, (int)alarm
    );

    delay(500);
}
